10RedisClient::RedisClient(const ConnectionOptions& options)

12{
13 try {
14 // Build connection string
15 sw::redis::ConnectionOptions conn_opts;
16 conn_opts.host = options_.host;
17 conn_opts.port = options_.port;
18 conn_opts.password = options_.password;
19 conn_opts.db = options_.db;
20 conn_opts.connect_timeout = options_.connect_timeout;
21 conn_opts.socket_timeout = options_.socket_timeout;
22
23 // Connection pool options
24 sw::redis::ConnectionPoolOptions pool_opts;
25 pool_opts.size = options_.pool_size;
26
27 // Create Redis client with connection pooling

30 // Test connection

34 }
35 catch (const sw::redis::Error& e) {
